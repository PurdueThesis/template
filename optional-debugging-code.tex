\ProvidesFile{optional-debugging-code.tex}[2022-05-18 optional debugging code]

%
%  This is a trick Mark Senn uses to to override the
%     \def\ZZprogram{...}
% statement for testing.
%
% If pdflatex, lualatex, etc. are called with, for example,
%     lualatex "\def\ZZOverrideProgram{eaps} \input{thesis}"
% or
%     lualatex "\def\ZZOverrideProgram{ece} \input{thesis}"
% then \ZZprogram is set to
%     Earth, Atmospheric, and Planetary Sciences
% or
%     Electrical and Computer Engineering
% respectively.
%
% If pdflatex,lualatex, etc. is called with
%     lualatex thesis
% then \ZZprogram is not changed from what is specified in thesis.tex.

% Override default values in thesis.tex file.
\def\ZZshowcolophon{true}
\def\ZZshowdiagonalline{true}
\def\ZZshowgridlines{true}
\def\ZZshowtimestamp{true}
\def\ZZtodonotes{wide}

\ExplSyntaxOn

  {

    % From answer 7.5 on page 308 of The TeXbook:
    %     If you type ‘\message{\string~}’ and ‘\message{\string\~}’, TEX
    %     responds with ‘~’ and ‘\~’, respectively. To get \12 from \string you
    %     therefore need to make backslash an active character. One way to do
    %     this is
    %         {\catcode‘/=0 \catcode‘\\=13 /message{/string\}}
    %     (The “null control sequence” that you get when there are no tokens
    %     between \csname and \endcsname is not a solution to this exercise,
    %     because \string converts it to ‘\csname\endcsname’. There is, however,
    %     another solution: If TEX’s \escapechar parameter---which will be
    %     explained in one of the next dangerous bends---is negative or greater
    %     than 255, then ‘\string\\’ works.)
    \escapechar = -1
    
    \immediate\write16{ENTERING~optional-debugging-code.tex.}

    \immediate\write16{\space\space \string\\ZZprogram~is~`\ZZprogram'.}
    
    \cs_if_exist:NTF
      \ZZOverrideProgram
      {
        \immediate\write16{\space\space \string\\ZZOverrideProgram~is~`\ZZOverrideProgram'.}
        \str_case:VnTF
          \ZZOverrideProgram
          {
            { eaps }  { \immediate\gdef\ZZprogram{Earth,~Atmospheric,~and~Planetary~Sciences} }
            { ece }   { \immediate\gdef\ZZprogram{Electrical~and~Computer~Engineering} }
          }
          {}
          {
            \immediate\write16{\space\space \string\\ZZOverrideProgram~(`\ZZOverrideProgram')~is~not~`eaps',~or~`ece'.}
            \stop
          }
      }
      {
        \immediate\write16{\space\space \string\\ZZOverrideProgram~is~not~set.}
      }


    \immediate\write16{\space\space \string\\ZZprogram~is~now~`\ZZprogram'.}

    \immediate\write16{LEAVING~optional-debugging-code.tex.}

  }

\ExplSyntaxOff
